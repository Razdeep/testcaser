139  explicit RandomIntegerLimit(long long upper, long long lower)
140  : UpperLimit(upper), LowerLimit(lower) {
141  if (lower >= upper) {
142  auto ex = testcaser::exceptions::maker::LimitError(
143  "Upper limit was : " + std::to_string(upper) +
144  "while lower limit was " + std::to_string(lower));
145  ex.set_resolution(
146  "Lower limit must be strictly smaller than upper limit. Try to "
147  "switch them to fix it");
148  throw ex;
149  }
150  }
159  void add_interval_exception(std::pair<long long, long long> interval) {
160  long long upper =
161  interval.first > interval.second ? interval.first : interval.second;
162  long long lower = interval.first + interval.second - upper;
163  if (upper > UpperLimit || lower < LowerLimit) {
164  throw testcaser::exceptions::maker::LimitIntervalError(
165  "Intervals upper limit is : " + std::to_string(upper) +
166  " while Limit's upper limit is " + std::to_string(UpperLimit) +
167  " and for lower " + std::to_string(lower) + " and " +
168  std::to_string(LowerLimit) + " respectively.");
169  }
170  except_intervals.emplace_back(std::make_pair(lower, upper));
171  };
177  unsigned long long actual_limit_size() {
178  long long min, max;
179  unsigned long long total_size = UpperLimit - LowerLimit;
180  if (except_intervals.size() == 0) return total_size;
181  std::sort(except_intervals.begin(), except_intervals.end());
182  min = except_intervals[0].lower;
183  max = except_intervals[except_intervals.size() - 1].upper;
184  total_size -= max - min;
185  auto active = except_intervals[0];
186  for (int t = 1; t < except_intervals.size(); t++) {
187  if (except_intervals[t].lower >= active.upper)
188  total_size += except_intervals[t].lower - active.upper;
189  if (active.upper < except_intervals[t].upper)
190  active = except_intervals[t];
191  }
192  return total_size;
193  };
202  bool valid_output(long long out) const {
203  if (out >= UpperLimit || out < LowerLimit) return false;
204  for (auto& e : except_intervals)
205  if (out >= e.lower && out < e.upper) return false;
206  return true;
207  };
